    ADD COLUMN IF NOT EXISTS finalizado_em TIMESTAMP NULL,
    ADD COLUMN IF NOT EXISTS duracao_ms INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS quantidade_operacoes INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS operacoes_executadas INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS operacoes_com_sucesso INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS operacoes_com_falha INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS status_http INT NOT NULL DEFAULT 0,
    ADD COLUMN IF NOT EXISTS sucesso BOOLEAN NOT NULL DEFAULT FALSE;
